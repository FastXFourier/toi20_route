นายสมหวังผู้ได้รับมอบหมายให้จัดรถขนส่งไฟฟ้า ได้พบว่า หากเราทราบว่ารถคันไหนวิ่งระหว่างจุดไหน เราสามารถคำนวณ “การจัดสรรแบบประหยัด” ได้ ซึ่งการจัดสรรแบบประหยัดนั้น คือ การเลือกรถ N−1N - 1N−1 คันจากเส้นทางที่กำหนดให้ เพื่อให้ค่าใช้จ่ายรวมของรถที่เลือกมีค่าน้อยที่สุด และผู้ใช้บริการจะต้องสามารถเดินทางไป-มาระหว่างจุดทั้ง NNN จุดได้

Summary

  • ใช้หลักการทำงานของ MST มาเลือกว่า จะให้ edge ไหน มี weight เท่าไหร่

Solution Code:

#include <bits/stdc++.h> 

using namespace std; 

vector <pair <int, int>> route(int n, vector <int> w) { 
	int m = w.size(); 
	vector <pair <int, int>> ans; 
	bool cant = false; 
	int curr = 1; 
	if (m == n && n >= 3) ans.emplace_back(1, 3); 
	if (m > n) { 
		for (int i = 2; i <= n; i++) { 
			ans.emplace_back(i, i - 1); m--; 
			for (int j = i - 2; j > 0; j--) { 
				if (m == n - i) { 
					cant = true; 
					break; 
				} 
				m--; 
				ans.emplace_back(i, j); 
			} 
			if (cant) { 
				curr = i; 
				break; 
			} 
		} 
	} 
	for (auto &[x, y] : ans) if (x > y) swap(x, y); 
	for (int i = curr; i < n; i++) ans.emplace_back(i, i + 1); 
	return ans; 
}

Total Time Complexity: O(M)O(M)O(M)
